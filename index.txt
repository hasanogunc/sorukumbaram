    <script>
        /**
         * Basit Uygulama Mantƒ±ƒüƒ± (Vanilla JS)
         */
        const app = {
            data: {
                username: 'Misafir',
                questions: []
            },
            currentImageBase64: null, // Y√ºkleme sƒ±rasƒ±ndaki ge√ßici g√∂rsel
            currentDetailId: null,

            // Ba≈ülangƒ±√ß
            init: function() {
                this.loadData();
                
                // Eƒüer kullanƒ±cƒ± adƒ± varsa direkt Home'a git, yoksa Onboarding
                if (localStorage.getItem('sk_username')) {
                    this.data.username = localStorage.getItem('sk_username');
                    this.router('home');
                } else {
                    this.router('onboarding');
                }
            },

            // Sayfa Y√∂nlendirme
            router: function(viewId) {
                // T√ºm view'larƒ± gizle
                document.querySelectorAll('.view').forEach(el => el.classList.remove('active'));
                
                // ƒ∞lgili view'ƒ± a√ß
                const target = document.getElementById(`view-${viewId}`);
                if (target) {
                    target.classList.add('active');
                    window.scrollTo(0,0);
                }

                // View a√ßƒ±lƒ±rken yapƒ±lacak √∂zel i≈üler
                if (viewId === 'home') this.updateHomeUI();
                if (viewId === 'archive') this.renderArchiveList();
                if (viewId === 'add') this.resetAddForm();
            },

            // Onboarding Tamamlama
            finishOnboarding: function() {
                const inputName = document.getElementById('input-username').value.trim();
                this.data.username = inputName || 'Misafir';
                localStorage.setItem('sk_username', this.data.username);
                this.saveData();
                this.router('home');
            },

            // Home Ekranƒ±nƒ± G√ºncelleme
            updateHomeUI: function() {
                document.getElementById('display-username').textContent = this.data.username;
                document.getElementById('total-count').textContent = this.data.questions.length;
            },

            // Dosya Se√ßimi ve Sƒ±kƒ±≈ütƒ±rma (√ñNEMLƒ∞: LocalStorage kotasƒ± i√ßin)
            handleFileSelect: function(input) {
                if (input.files && input.files[0]) {
                    const file = input.files[0];
                    const reader = new FileReader();
                    
                    reader.onload = (e) => {
                        // Resmi sƒ±kƒ±≈ütƒ±rmak i√ßin Image objesi olu≈ütur
                        const img = new Image();
                        img.src = e.target.result;
                        
                        img.onload = () => {
                            // Canvas olu≈ütur
                            const canvas = document.createElement('canvas');
                            const ctx = canvas.getContext('2d');
                            
                            // Yeni boyutlarƒ± hesapla (Max width 800px)
                            const maxWidth = 800;
                            let width = img.width;
                            let height = img.height;
                            
                            if (width > maxWidth) {
                                height = height * (maxWidth / width);
                                width = maxWidth;
                            }
                            
                            canvas.width = width;
                            canvas.height = height;
                            
                            // Resmi canvas'a √ßiz
                            ctx.drawImage(img, 0, 0, width, height);
                            
                            // Base64 string'e √ßevir (JPEG, 0.7 kalite)
                            const compressedDataUrl = canvas.toDataURL('image/jpeg', 0.7);
                            
                            // √ñnizleme ve state g√ºncelleme
                            document.getElementById('preview-image').src = compressedDataUrl;
                            document.getElementById('preview-image').style.display = 'block';
                            document.querySelector('.preview-placeholder').style.display = 'none';
                            
                            app.currentImageBase64 = compressedDataUrl;
                        };
                    };
                    
                    reader.readAsDataURL(file);
                }
            },

            // Soru Kaydetme
            saveQuestion: function(e) {
                e.preventDefault();

                // Validasyon (HTML5 required zaten engelliyor ama ekstra kontrol)
                if (!this.currentImageBase64) {
                    alert("L√ºtfen bir fotoƒüraf se√ßin.");
                    return;
                }

                const lesson = document.getElementById('input-lesson').value;
                const status = document.querySelector('input[name="status"]:checked').value;
                const note = document.getElementById('input-note').value;

                const newQuestion = {
                    id: Date.now(), // Unique ID
                    lesson: lesson,
                    status: status,
                    note: note,
                    image: this.currentImageBase64,
                    createdAt: new Date().toISOString()
                };

                this.data.questions.unshift(newQuestion); // En ba≈üa ekle
                this.saveData();

                this.showToast("Soru Ar≈üivlendi! ‚úÖ");
                this.router('home');
            },

            // Formu Temizle
            resetAddForm: function() {
                document.querySelector('form').reset();
                this.currentImageBase64 = null;
                document.getElementById('preview-image').style.display = 'none';
                document.querySelector('.preview-placeholder').style.display = 'block';
            },

            // Ar≈üiv Listeleme
            renderArchiveList: function() {
                const listContainer = document.getElementById('archive-list');
                const filterLesson = document.getElementById('filter-lesson').value;
                
                listContainer.innerHTML = '';

                // Filtreleme
                let filtered = this.data.questions;
                if (filterLesson !== 'all') {
                    filtered = filtered.filter(q => q.lesson === filterLesson);
                }

                if (filtered.length === 0) {
                    listContainer.innerHTML = '<div style="text-align: center; color: #999; margin-top: 30px;">Bu kriterde soru bulunamadƒ±.</div>';
                    return;
                }

                // Kartlarƒ± Olu≈ütur
                filtered.forEach(q => {
                    const statusTextMap = { 'yanlis': 'Yanlƒ±≈ü', 'bos': 'Bo≈ü', 'zor': 'Zorlandƒ±' };
                    const statusClassMap = { 'yanlis': 'badge-yanlis', 'bos': 'badge-bos', 'zor': 'badge-zor' };
                    
                    const date = new Date(q.createdAt).toLocaleDateString('tr-TR', { day: 'numeric', month: 'short', hour: '2-digit', minute: '2-digit' });

                    const card = document.createElement('div');
                    card.className = 'card';
                    card.onclick = () => app.openDetail(q.id);
                    
                    card.innerHTML = `
                        <img src="${q.image}" class="card-thumb" loading="lazy">
                        <div class="card-content">
                            <div style="display:flex; justify-content:space-between; align-items:flex-start;">
                                <div style="font-weight:700; color:#333;">${q.lesson}</div>
                                <div style="font-size:12px; color:#888;">${date}</div>
                            </div>
                            <span class="card-badge ${statusClassMap[q.status]}">${statusTextMap[q.status]}</span>
                            ${q.note ? `<div style="font-size:12px; color:#666; margin-top:5px; white-space:nowrap; overflow:hidden; text-overflow:ellipsis; max-width:200px;">üìù ${q.note}</div>` : ''}
                        </div>
                    `;
                    listContainer.appendChild(card);
                });
            },

            // Detay A√ßma
            openDetail: function(id) {
                const q = this.data.questions.find(item => item.id === id);
                if (!q) return;

                this.currentDetailId = id;
                const container = document.getElementById('detail-content');
                const statusTextMap = { 'yanlis': 'Yanlƒ±≈ü', 'bos': 'Bo≈ü', 'zor': 'Zorlandƒ±' };
                const date = new Date(q.createdAt).toLocaleString('tr-TR');

                container.innerHTML = `
                    <img src="${q.image}" class="detail-image">
                    
                    <div class="detail-meta">
                        <div class="detail-row">
                            <span>Ders</span>
                            <span>${q.lesson}</span>
                        </div>
                        <div class="detail-row">
                            <span>Durum</span>
                            <span>${statusTextMap[q.status]}</span>
                        </div>
                        <div class="detail-row">
                            <span>Tarih</span>
                            <span>${date}</span>
                        </div>
                    </div>

                    ${q.note ? `
                    <div class="detail-note">
                        <strong>Notum:</strong><br>
                        ${q.note}
                    </div>` : ''}
                `;

                // Silme butonuna event
                document.getElementById('btn-delete').onclick = () => {
                    if (confirm('Bu soruyu ar≈üivden silmek istediƒüine emin misin?')) {
                        app.deleteQuestion(id);
                    }
                };

                this.router('detail');
            },

            // Soru Silme
            deleteQuestion: function(id) {
                this.data.questions = this.data.questions.filter(q => q.id !== id);
                this.saveData();
                this.showToast('Soru silindi.');
                this.router('archive');
            },

            // LocalStorage ƒ∞≈ülemleri
            saveData: function() {
                localStorage.setItem('sk_data', JSON.stringify(this.data.questions));
            },

            loadData: function() {
                const stored = localStorage.getItem('sk_data');
                if (stored) {
                    try {
                        this.data.questions = JSON.parse(stored);
                    } catch (e) {
                        console.error('Veri okuma hatasƒ±', e);
